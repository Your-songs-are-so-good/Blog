浅析 async/await 异步函数

async/await 异步函数可以将异步行为变成同步执行，并返回一个promise

async function asyncFunction () {
    const value = await Promise.resolve(2);
    console.log('value:', value);     // value: 2
    try {
        // 错误的 promise 会抛出异常，需要 try...catch 捕获
        const value = await Promise.reject(3);
    } catch (reason) {
        console.log('reason:', reason);  // reason: 3
    }
    // 没有返回值，默认是解决状态，值为 undefined
}
